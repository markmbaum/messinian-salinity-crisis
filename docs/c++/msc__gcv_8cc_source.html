41 void MscGcv::ode_fun (double *solin, double *fout) {
42 
43  double zs, zm, zo, Am, tau, S, Q, T;
44  //label incoming sill and Mediterranean levels
45  zs = solin[0];
46  zm = solin[1];
47  //Mediterranean area and ocean level are functions of zm
48  Am = fAm(zm);
49  zo = fzo(zm);
50  //channel slope
51  /*NOTE:
52  The slope is not allowed to be negative here for numerical reasons.
53  If the model behaves physically, the slope should never be negative because
54  the flow velocity over the sill goes to zero at exactly zo = zm. The basin
55  should never fill above the ocean, not even by a tiny amount. In some cases,
56  however, a tiny negative slope appears somewhere in the time stepping
57  process. This is almost certainly just a numerical inaccuracy that occurs
58  when the Mediterranean is filling extremely rapidly. However, if S is ever
59  a negative number, even a vanishingly small one, it blows up the
60  calculation of Q upon the evaluation of S^(13/14). When a NaN appears
61  there, it carries through all subsequent calculations and the model fails.
62  So, even though it should not be necessary on a physical basis, negative
63  slope values are forced to zero for numerical feasibility. This issue
64  cropped up in lots of different ODE solving algorithms, explicity and
65  implicit, so it is not specific to the algorithm used here.*/
66  S = max((zo - zm)/L, 0.0);
67  //sill shear stress
68  tau = RHO*GRAV*(zo - zs)*S;
69  //bundle of constants
70  T = Cw*pow((tauc + U/kb)/(RHO*GRAV), -3.0/13);
71  //discharge for Turowski width
72  Q = (pow(T, 13.0/7)/n)*pow(max(zo - zs, 0.0), 65.0/21)*pow(S, 13.0/14);
73  //sill level time derivative, accounting for the critical Ï„ threshold
74  fout[0] = (tau > tauc) ? U - kb*pow(tau - tauc, a) : U;
75  //Mediterranean level time derivative
76  fout[1] = P - E + (R + Q)/Am;
77 }

180 double MscGcv::index (double zs, double zm, double r) {
181 
182  double tol = 3.0*PI/4.0;
183  //array for angle of vector field

187  double theta;

192  theta = 2*PI*double(i)/double(n-1);
193  //coordinates of sample point
194  z[0] = zs + r*cos(theta);
195  z[1] = zm + r*sin(theta);
196  //vector components
197  ode_fun(z, v);
198  //angle of components
199  phi[i] = atan2(v[1], v[0]);
200  }
201  //correct for jumps
202  double jump;
203  for (long i=1; i<n; i++) {
204  if ( (phi[i] > tol && phi[i-1] < -tol) || (phi[i] < -tol && phi[i-1] > tol) ) {
205  jump = (phi[i-1] > 0.0) ? 2*PI : -2*PI;
206  for (long j=i; j<n; j++) phi[j] += jump;
207  }
208  }
209 
210  return( (phi[n-1] - phi[0])/(2*PI) );
